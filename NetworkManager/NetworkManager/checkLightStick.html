Checks if the device at the given IP address is a light stick.

If it is, it adds the device to the list of active devices.

static void checkLightStick(String deviceIp) async{
  var url = Uri.parse("http://$deviceIp/wifi");
  var client = http.Client();
  var response = await client.get(url);
  String output = response.body.toString();
  print(output);
  if(output.contains("WIFI")) {
    String hostname = output.split(": ").last;
    if(!activeDevices.containsKey(deviceIp)) {
      activeDevices[deviceIp] = Device(ipAddress: deviceIp, hostname: hostname);
      //make it in order of hostname
      activeDevices = Map.fromEntries(
          activeDevices.entries.toList()..sort((e1, e2) => e1.value.hostname.compareTo(e2.value.hostname))
      );
      refreshHomePage();
      print("done");
    }
  }
}
