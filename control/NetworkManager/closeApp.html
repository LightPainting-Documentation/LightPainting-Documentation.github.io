Future<void> closeApp()

Closes the UDP receiver and sends a 'resendIP' request to each active device to reset the device IP

This function is called when the app is paused or closed by the user

Implementation

static Future<void> closeApp() async{
  receiver?.close();
  activeDevices.forEach((deviceIp, value) async{
    try{
      var url = Uri.parse("http://$deviceIp/resendIP");
      var response = await http.post(
        url,
        headers: <String, String>{
          'Content-type': 'application/json',
        },
      );
      if (response.statusCode == 200){
        print("$deviceIp done");
      }
    }
    catch(e){
      print(e);
    }
  });
}
