void drawPaintImage(Canvas canvas) async{
  canvas.translate(0, imageToDraw.image.width/2);
  canvas.rotate(-90* pi /180);
  canvas.save();
  paintImage(
    flipHorizontally : isReversed,
    canvas: canvas,
    rect: Rect.fromCenter(
        center: const Offset(0,0),
        width:  imageToDraw.image.width.toDouble(),
        height: imageToDraw.image.height.toDouble()
    ),
    image: imageToDraw.image,
  );

  if(stopPixel == -1){
    stopPixel = imageToDraw.image.height.toDouble();
  }

  // Draw a rectangle to represent the start of the image.
  canvas.drawRect(
      Rect.fromLTWH(-imageToDraw.image.width/2, -imageToDraw.image.height/2, imageToDraw.image.width.toDouble(), startPixel),
      Paint()
  ); //start

  // Draw a rectangle to represent the end of the image.
  canvas.drawRect(
      Rect.fromLTWH(- imageToDraw.image.width/2, (imageToDraw.image.height/2) - (imageToDraw.image.height - stopPixel) + 1, imageToDraw.image.width.toDouble(), (imageToDraw.image.height - stopPixel - 1)),
      Paint()
  ); //end
  canvas.restore();
}
