void displayAlert({
  required String message,              // Alert message to be displayed
  String header = "Success",            // Header text for the alert, by default it is "Success"
  bool noDevicesInfo = false,           // Boolean to display open hotspot settings button
  required BuildContext context         // current buildcontext to display Alert message
}) {
  showDialog<String>(
    context: context,
    builder: (BuildContext context) => AlertDialog(
      title: Text(
        header,
        style: const TextStyle(
            color: Colors.white70
        ),
      ),
      content: Text(message),
      actions: <Widget>[
        // If noDevicesInfo is true, display a button to open the hotspot settings
        noDevicesInfo ?
        TextButton(
          child: const Text("Open Hotspot Settings"),
          onPressed: () {
            Navigator.pop(context, 'OK');
            SystemSettings.wireless();
          },
        ) :
        const SizedBox(),
        // Display an "OK" button to close the dialog
        TextButton(
          onPressed: () => Navigator.pop(context, 'OK'),
          child: const Text('OK'),
        ),
      ],
    ),
  );
}
